k = 850001
n = 0
while(n != 6):    # пока не найдётся 6 чисел, n - это кол-во чисел найденных
    delitel = []  # список делителей каждого числа
    f = 0         # число F
    for i in range(2, k):   # перебираем все числа от 2 до самого числа, проверяем на делимость
        if (k % i == 0):
            delitel.append(i)
    sorted(delitel)    # сортируем для удобства список с делителями
    if(len(delitel) != 0):   # проверяем, что делители есть
        f = delitel[len(delitel) - 1] - delitel[0]   # находим число F
    if(f != 0 and f % 7 == 0):  # проверяем, что число нашлось и что оно делится на 7
        n += 1
        print(f"Число:{k}, его F: {f}")
    k += 1
