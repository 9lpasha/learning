filename = 'C:/Users/13131/Downloads/XYZ.txt'
with open(filename) as file_object:
    lines = file_object.readlines()
# открыли файл и считали все строки символов XYZ, но там одна строка, так что считали одну строку в список lines,
# получается в списке lines только один элемент - считанная строки
i = 1   # местоположение конкретного символа в большой считанной строке
max = 0   # максимальное кол-во подряд идущих букв в алфавитном порядке
k = 0   # это положение последнего символа в строке подряд идущих в алфавитном порядке букв
s = 1   # кол-во подряд идущих букв в алфавитном порядке
for line in lines:
    while(i < len(line)):
        if(line[i - 1] == 'X'):   # за X может идти либо X, либо Y
            if(line[i] == 'Y' or line[i] == 'X'):
                s +=1   # продолжаем строку если условие истинно
            else:   # в противном случае строим строку заново (s = 0), и если предыдущая строка была длиннее всех, то записываем её результат
                if(s > max):
                    max = s
                    k = i
                s = 1
        if (line[i - 1] == 'Y'):   # то же самое с Y и Z
            if (line[i] == 'Y' or line[i] == 'Z'):
                s += 1
            else:
                if (s > max):
                    max = s
                    k = i
                s = 1
        if (line[i - 1] == 'Z'):
            if (line[i] == 'Z'):
                s += 1
            else:
                if (s > max):
                    max = s
                    k = i
                s = 1
        i += 1
print(max, k)
